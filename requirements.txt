# Runtime dependencies for the Function Compute bundle (TASK-003).
#
# `s build` installs these into the code directory before `s deploy` uploads it. Without this file
# the bundle is source-only and the function dies at cold start on `import fastapi` - which is
# exactly how the first deploy failed.
#
# This mirrors pyproject's core + [agents,memory,cloud,parse,pdf] extras. It deliberately omits the
# dev toolchain and the obs extra: a fatter bundle is a slower cold start on every single request,
# and the OTel SDK is optional by design (obs/otel.py no-ops without it).

pydantic>=2.9
pydantic-settings>=2.7
fastapi>=0.115
python-multipart>=0.0.9
# FC's Python runtime serves HTTP functions over WSGI; FastAPI is ASGI. This adapts one to the
# other. Without it the function deploys cleanly and then 502s on every request with no stack
# trace - which is exactly how the first deployment failed.
a2wsgi>=1.10
python-ulid>=3.0
PyYAML>=6
jinja2>=3.1

# Agents (AgentScope + the DashScope OpenAI-compatible client)
agentscope==1.0.9
openai>=1.55,<3

# Memory + durable state
tablestore>=6.4.7
tablestore-for-agent-memory==1.1.3

# Object storage
oss2>=2.19,<3
certifi

# pyOpenSSL and cryptography are pinned together, and that pairing is load-bearing.
#
# oss2 imports aliyunsdkcore, which imports a vendored urllib3 that imports pyOpenSSL. The FC
# runtime ships its own (old) pyOpenSSL in /var/fc/.../site-packages, but our bundle at /code/python
# comes first on the path and shadows the runtime's `cryptography` with a modern one. Modern
# cryptography removed the `GEN_EMAIL` binding that old pyOpenSSL still dereferences at import time,
# so the function died at cold start with:
#
#     AttributeError: module 'lib' has no attribute 'GEN_EMAIL'
#
# Shipping pyOpenSSL ourselves means both halves of the pair come from the same place and agree.
# Pinning only one of them is what caused the collision in the first place.
pyOpenSSL>=24.2
cryptography>=42

# Parsing: spreadsheets, digital PDF text, and scan rasterisation (TASK-031/032)
openpyxl>=3.1
pypdfium2>=5.9
pillow>=10

# PDF rendering (TASK-090). NOTE: WeasyPrint needs native pango/cairo. It is imported lazily inside
# render_pdf, so the function boots and serves the API without it; if the runtime lacks the system
# libraries, PDF generation is the only thing that fails, and it fails loudly rather than silently.
weasyprint==68.*
